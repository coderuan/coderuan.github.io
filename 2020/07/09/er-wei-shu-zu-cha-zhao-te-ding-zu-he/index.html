二维数组查找特定组合

算法 笔记
JavaSE
发布日期:   2020-07-09
文章字数:   2.6k

舍友面试看见的一道算法题

public class Count {
    //定义要查询的查询组合
    public static char[] apple = {'a','p','p','l','e'};
    //记录查询的下标
    public static int index = 0;
    //记录次数
    public static int num = 0;

    public static void main(String[] args) {
        //题目要求查询的二维数组
        char[][] c = {
                {'a','p','c','d','h','x'},
                {'i','p','p','a','d','b'},
                {'e','l','p','a','d','b'},
                {'e','l','c','a','d','b'}
            };
        //开始查找
        find(c);
    }

    private static void find(char[][] c) {
        // 1.首先遍历所有,找出a
        for (int i = 0; i < c.length; i++) {
            for (int j = 0; j < c[i].length; j++) {
                if(c[i][j]==apple[index]) {
                    // 2.从a开始,遍历周围,查找下一个目标字符
                    findChar(-1,-1,i,j,c,apple[++index]);
                    index = 0;//每次查找完a,清空下标
                    System.out.println("--------- 一个'a'查找完毕 -----------");
                }
            }
        }
        System.out.println("总次数："+num);
    }

    //定义找到的 apple[index] 字符坐标为 (xEnd,yEnd)
    //它的上一个坐标为 (xStart,yStart) ————>防止重复查找
    private static void findChar(int xStart, int yStart,int xEnd,int yEnd, char[][] c,char target) {
        if (xEnd==0) {    //1、在第一行
            for (int i=xEnd; i<=(xEnd+1) ; i++) {    //遍历本行以及下一行
                loopY(xStart,yStart,xEnd, yEnd, i, c,target);
            }
        } else if( xEnd==(c.length-1) ){    //2、在最后行
            for (int i=(xEnd-1); i<=xEnd; i++) {    //遍历本行以及上一行
                loopY(xStart,yStart,xEnd, yEnd, i, c,target);
            }
        } else {    //3、在中间行
            for (int i=(xEnd-1); i<=(xEnd+1); i++) {    //遍历本行及上下两行
                loopY(xStart,yStart,xEnd, yEnd, i, c,target);
            }
        }
    }

    //判断列的位置是否在边缘，然后分情况循环查询
    public static void loopY(int xStart, int yStart,int xEnd, int yEnd,int i, char[][] c,char target) {
        if(yEnd==0) {    //在第一列
            for (int j = yEnd; j<=(yEnd+1); j++) {    //遍历本列以及右一列
                check(xStart, yStart, xEnd,yEnd,i,j,c,target);//符合条件，下一轮寻找
            }
        }else if( yEnd==(c[xEnd].length-1) ) {    //在最后列
            for (int j = (yEnd-1); j<=yEnd ; j++) {    //遍历本列以及左一列
                check(xStart, yStart, xEnd,yEnd,i,j,c,target);//符合条件，下一轮寻找
            }
        }else {    //在中间列
            for (int j = (yEnd-1); j<=(yEnd+1) ; j++) {    //遍历本列，及左右列
                check(xStart, yStart, xEnd,yEnd,i,j,c,target);//符合条件，下一轮寻找
            }
        }
    }

    //开始检验，符合条件回调继续查询
    public static void check(int xStart, int yStart,int xEnd, int yEnd,int i,int j, char[][] c,char target) {
        if(xStart==i && yStart==j) return;    //排除上一个点
        if(xEnd==i && yEnd==j) return;    //排除自身
        if(c[i][j]==target) {
            System.out.printf("[debug]从第%d行,第%d列————>找到 第%d行,第%d列: %c 字母.\n",
                    (xEnd+1),(yEnd+1),(i+1),(j+1),target);
            if(target == apple[apple.length-1]) {    //找到目标
                System.out.printf("[debug]第%d次找到目标e\n",num+1);
                num++;    //计数加一
                return;
            }
            findChar(xEnd,yEnd,i,j,c,apple[++index]);//符合条件，下一轮寻找
            --index;    //找完之后，回到上一个节点，继续找
        }
    }

}

总结

  • 不能空想，要模仿计算机，写出执行步骤，判断逻辑。
  • 可别再复制黏贴了，就一个 x 和 y 写混了，改了一早上，佛了。
  • 多写注释，不然过了一晚上，自己的代码都看懵了。

其实本次实现还不够完善，因为字符里面只有2个 ‘p’ 是重复的，偷懒省去了很多判断。

上面的代码只取了上一个节点的坐标，保证2个 ‘p’ 没有重复，没有判断的5个点的坐标。
